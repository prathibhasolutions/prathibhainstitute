Switch Case

21. Write a C program to create a simple calculator using switch statement that can perform basic arithmetic operations: addition, subtraction, multiplication, and division
#include <stdio.h>
int main() {
    char op = '+';
    int a = 10, b = 5;
    switch(op) {
        case '+': printf("Sum = %d", a + b); break;
        case '-': printf("Difference = %d", a - b); break;
        case '*': printf("Product = %d", a * b); break;
        case '/': printf("Quotient = %d", a / b); break;
        default: printf("Invalid operator");
    }
    return 0;
}

33. Write a C program to print a pyramid pattern of stars (*) using nested loops.
#include

    int rows = 5;
    for(int i = 1; i <= rows; i++) {

34. Write a C program to print an inverted pyramid pattern of stars (*) using nested loops.
#include 
int main() {
    int rows = 5;
    for(int i = rows; i >= 1; i--) {

        for(int k = 1; k <= 2*i - 1; k++) printf("* ");
        printf("\n");
    }
    return 0;
}

64. Write a C program that defines a function to check whether a given integer is a prime number. The function should take an integer as an argument and return 1 if the number is prime, otherwise return 0. Call this function from main() to check if a user-input number is prime and print the result. 
#include 
int isPrime(int n) {
    if(n < 2) return 0;
    for(int i = 2; i <= n/2; i++) {
        if(n % i == 0) return 0;
    }
    return 1;
}
int main() {
    int num = 7;
    if(isPrime(num)) printf("%d is prime", num);
    else printf("%d is not prime", num);
    return 0;
}
65. Write a C program that uses a function to generate and print the Fibonacci series up to n terms. The function should take an integer n as an argument and print the Fibonacci sequence up to n terms. Call this function from main() after taking input from the user.
#include 
void fibonacci(int n) {
    int a = 0, b = 1, next;
    for(int i = 0; i < n; i++) {
        printf("%d ", a);
        next = a + b;
        a = b;
        b = next;
    }
}
int main() {
    int terms = 5;
    fibonacci(terms);
    return 0;
}
